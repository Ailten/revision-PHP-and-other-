    public function up(): void  // when up (start).
    {
        Schema::create('users', function (Blueprint $table) {  // create the table "users".
            $table->id();   // set a param named "id".
            $table->string('email')->unique();  // param name "email", type string, and unique contraint.
            $table->string('email', 64);  // set length char.
            $table->integer('int')->unsigned();  // integer unsinged.
            $table->float('float');  // float.
            $table->tinyInteger('byte');  // from -128 to 127.

            $table->timestamps();  // add 2 time stamps, one who edit eatch time the object is update on DB, and one when it create in DB.
            $table->foreignId('user_id')->constrained()->onDelete('cascade');  // add a FK to user 
            // need "constrained()", with param "user" (table target), without param it take the name befor "_id".
            // "onDeletete('cascade')" say "delete this OBJ when the FK is deleted".
        });
    }
    public function down(): void  // when down (rollback).
    {
        Schema::dropIfExists('users');  // drop the talbe "users".
        Schema::table('orders', function(Blueprint $table){  // block to delete cascade FK constraint, write it on table who has the FK.
            $table->dropForeignIdFor(User::class);  // class form model.
        });
    }
}

// make a Validator (Illuminate\Support\Facades).
$validator = Validator::make([
    'title' => 'value'  // user send a title empty, at creating the obj Model.
], [
    'title' => 'required',  // define parameter title as contraint "required" (throw error if not sended).
    'value' => 'numeric'  // numeric axept number (entier or decimal).
    'value' => 'integer'  // axept only number entier.
    'value' => 'string'  // refuse number, array or object.
    'value' => 'integer|min:1',  // heer, define integer make the "min,max" define on the value, not on the length str.
    'email' => ['required', 'min:8'],  // heer, min tel the champ need to be 8 char length min.
    'abc' => ['regex:/^[a-z]+$/']  // regex example.
    'role_id' => ['exists:roles,id']  // need to exist in table roles (specify what table, and what champ in this table).
    'name' => ['unique:users,name']  // verify if the champ is not already on a DB champ unique (users is table verify, name is the champ verify).
    'name' => [Rule::unique('users')],  // same, but with object rule (instead of string), heer unique in table 'users'.
    'name' => [Rule::unique('users', 'email')->ignore($user->id)]  // case of update (to igniore them self).
    'value' => 'not_in:0'  // make error if value is "0".
    'value' => [function($attribute, $value, $fail) { if($value == 0) { $fail('msgError); } }]  // declare the function error custom.
]);

// ---------------> (-) Eloquance, example for relation in model :

// proper way to create A and B (by using function definition of relation in model).
class B extends Model
{
    protected $fillable = ['name'];
    public function as()      // --- define "model B can have many A" (A has B_id as FK), (call "as" for "A" many).
    {
        return $this->hasMany(A::class);
        return $this->hasMany(A::class, 'b_id', 'id');  // redfine champs (fk and pk).
    }
}
class A extends Model
{
    protected $fillable = ['name'];
    public function b()       // --- define "model A have one B" (A has B_id as FK).
    {
        return $this->belongsTo(B::class);
        return $this->belongsTo(B::class, 'b_id', 'id');  // redefine champs (fk and pk).
    }
    public function cs() {    // --- define "model A have many C, from join-table B" (A-B-C).
        return $this->belongsToMany(C::class, 'bs');
    }
}
DB::transaction(function () {
    $b = B::create([
        'name' => 'Mon B'
    ]);
    $b->as()->create([    // --- use heer to create from B (create with $b as FK).
        'name' => 'Mon A'
    ]);
});

// on a class model A, to define execution additional when interact with db (crud).
protected static function booted()
{
    static::creating(function ($a) {
        DB::transaction(function () use ($a) {
            $b = B::create([]);
            $a->b_id = $b->id;
        });
    });
    static::deleting(function ($a) {
        $a->b->delete();
    });
}
public function characterSpecie() {  // for call $a->b.
    return $this->belongsTo(B::class);
}
