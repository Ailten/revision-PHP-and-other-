docker --help  // print all details of cmd docker.
// site docker hub, site for download image for VMs docker.
docker pull [image-name]:[num-version]  // download an image.
docker image ls  // list all images download.
docker run [image-name]  // execute a VM.
docker container run [container-id-in-docker-software]  // run a container (group of VM), from id (found on software).
docker run --name [myContenerName] [image-name]  // same as run, set a name (to axess more easily).
docker run -p [27017:27017] [image-name]  // same as run, set port (of base machine, to VM).
docker -compose up [-d]  // turn on a compose.
docker -compose down  // turn off a compose.
