
#!/bin/sh  // start a shell script (.sh).
# comment.  // comment.
sudo [cmd]  // execute a command in admin mode (for permission).
sudo apt update  // maj of packages disponible (install nothing).
sudo apt upgrade  // installe maj of package already installed.
sudo apt install [package-name]  // install a new package.
MY_VAR=test.txt  // make a var.
"$MY_VAR"  // use a var.
ctrl-C  // cut a process running.
vim, hx  // editor with console.
"a b"  // use " for srting with space.
*  // replace every char, 0~n times in a string.
>  // redirection, took return of left contend, and send to a file right.
|  // pipe, took return of left contend, and use it as input to right contend.
;  // combine two commande on a same line (execute text to, but on same line).

cd  // change directory (to racine user).
cd ../  // same but go back from one folder.
cd [path]  // same but send a path.
cd -  // go back to the last folder visited.

ls  // list all file/folder on the current folder.
ls -l  // same, more details.
ls [path]  // same, for a specified folder.
ls -l --full-time  // same, but with the time in full char.
ls -lu  // same, but print also the date of last axess to files (not the date when file was modified).

pwd  // return the path of current folder (print it).

touch [file]  // create a file.
touch [a] [b] [c]  // create many files.
touch [file]{1..9}  // create 9 files (name end by number).
touch [file]{a..z}  // same with letter.
touch [existing-file]  // update date acces and modificate (open and update).
touch -t 199901010000 [file]  // same but send the date.
touch -d tomorrow [file]  // same but sending a date day relative.
touch -a [file]  // actualise the date of axess file (not modifed), like open it.
touch -at 9901010000 [file]  // same but set the date.
touch -m [file]  // updat date modificate (update).
touch -mt 9901010000 [file]  // same but set the date.
touch -am [file]  // edit the date acces and modificate (open and update), (same as no option).
touch -r [file-ref] [file]  // set the date of a file to another one.
touch -h [file]  // update date of a symlink (symbolic link) (?).
touch -c [file]  // disable the "create if not exist" functionality.

mkdir [folder-name]  // make a new folder.
mkdir [a] [b] [c]  // make many folder.
mkdir -p [new-folder/new-folder]  // make new folders in cascade (create all).

rm [file-name]  // remove a file or a directory.
rm -r [folder-name]  // remove a folder recurcively (with all in it).
rm -rf [folder-name]  // -f for disable the confirmation request.

rmdir [folder-name]  // remove a folder (only empty), less permissiv, for user not-admin (?).
rmdir -p [folder-name]  // same, but force it.

cp [file-source] [path-dest]  // copy a file (or directory), at a folder path.
cp [file-source] [path-dest/new-name]  // same but with a rename.
cp -r [folder-source] [path-dest/]  // copy a folder (-r for recursive, took all the contend with).

mv [file-source] [path-dest]  // move a file (or directory), at a folder path.
mv [file-source] [path-dest/new-name]  // same but with rename (can be use to only rename).

cat [file]  // return the contend of a file (print it).
cat [a] [b] [c]  // return many contend files.
cat -n [file]  // same, but with number at eatch line.
cat > [new-file]  // create a new file (if not exist), and start writing on it (in console), ctrl-C to quite.
cat [a] [b] > [b]  // concat many file contend into a file (erase last contend).

less [file]  // return the contend of a file, with pagination (print it).
more [file]  // same but for files more big (not all tools).

tail [file]  // return the 10 last line of the file (print it).
tail -n 3  // same, but precise quantity of last line return.
head [file]  // same but for the first lines.
head -n 3  // same, but precise quantity of first line return.

grep [word-search] [file]  // return the lines find with the word searched on a file contend.
grep [w-s] [file-A] [file-B]  // same on many files.
grep -r [w-s] [folder]  // same but recurcive (aply on every files decending the folder).
grep -A3 [w-s] [file]  // took 3 line after the match.
grep -B3 [w-s] [file]  // took 3 line before the match (include the match them self).
grep -C3 [w-s] [file]  // took 3 line after AND before the match.
grep -i [w-s] [file]  // -i for disabled case sensitive.
grep -v [w-s] [file]  // -v for reverce search, return line who NOT match.
grep -n [w-s] [file]  // -n for add line number.
grep -E '^A+$' [file]  // -E to use regex.
.  // every char, one time.
*  // every char, zero to N time.
+  // last char, one to N time.
?  // last char, zero to one time.
^  // the start of an expression.
$  // the end of an expression.
()  // groupe motif.
grep 'w1\|w2\|w3' [file]  // search many words distinct.
grep -E 'w1|w2|w3' [file]  // same with regex.
grep -e 'w1' -e 'w2' [file]  // same with distinct parameters.
egrep [rgx] [file]  // same as -E. 
grep -c [w-s] [file]  // -c for return the amount of lines match.
grep -o [w-s] [file]  // -o for return the match only (not the line).
grep -o [w-s] [file] | wc -l  // return the count of match only.
grep -o '\bWordSearch\b' [file]  // return the matches exact (do not match if the word is a part of another one).
ls | grep [w-s]  // do grep on a return of ls.
grep [w-s] [file] > [file-save]  // return the result of grep on a file.

find -name "run*"  // return all file/folder having the name send (in current folder).
find -name "*.txt"  // same, find by extantion.
find [path] -name "run*"  // same, sending a folder.
find -type f -name "run*"  // -type f, for file only.
find -type d -name "run*"  // -type d, for directory only.
find -size +50M  // -size, for filter by size octe.
find -name "run*" -exec rm {} \;  // -exec, for execute a cmd on all result (heer is a rm) "{}" replace the result find.

ps  // return the list of process running currently (including theme self).
ps -A  // -A for add all process of the systeme.
ps -r  // -r for only process currently running (not those who sleep).
ps -u  // -u for add details of user who associate all process.

top  // open an analyse view of processus and performence systeme.
q  // for quite.
k  // for kill.
h  // for help.
M  // order by memory used.
P  // order by CPU used.

kill [id]  // end a process (without option, it save the progression of process).
kill -9 [id]  // same, without save (most used).
kill -1 [id]  // same, and re launch it.
kill -15 [id]  // same, end properly.
pkill -9 [process-name]  // kill with name (can kill many process at same time).

chmod -rwx---r-- [file]  // set permission to a file.
chmod -[.]---r-- [file]  // owner permissions.
chmod -rwx[.]r-- [file]  // group permissions.
chmod -rwx---[.] [file]  // other permissions.
chmod -rwx---r-- [owner-name] [users] [file]  // same, set owner-name and users-group.
chmod -x [file]  // remove execution permission to a file (without option, to all categorie users), like "a-x".
chmod u+x [file]  // add execution permission to a file (with option "u", to user categorie (owner)).
chmod g+r [file]  // add read permission to a file ("g" for group categorie).
chmod o-w [file]  // remove write permission to a file ("o" for other categorie).
chmod 711 [file]  // set permission to a file.
chmod .11 [file]  // owner permissions.
chmod 7.1 [file]  // group permissions.
chmod 71. [file]  // other permissions.
4, 2, 1  // 4 for read, 2 for write, 1 for execute.

chown [owner-name] [file]  // change the owner of a file.
chown :[group-name] [file]  // change the group of a file.
chgrp [group-name] [file]  // same.
chown [owner-name]:[group-name] [file]  // change both.

useradd [user-name]  // create an user.
useradd -m -s [user-name]  // same, with home and shell (folders).
passwd [user-name]  // set a password to an user.
groupadd [group-name]  // create a new group (users).
cat /etc/passwd  // list all users.
cut -d: -f1 /etc/passwd  // same, but only the name of users.
cat /etc/group  // list all group.
cut -d: -f1 /etc/group  // same, but only the name of group.
usermod -g [grp-name] [usr-name]  // set a main-group to an user.
usermod -aG [grp-name] [usr-name]  // set a second-group to an user.
-g  // main-group.
-G  // second-group.
-a  // add group (without "a" it replace).
usermod -G [grp-1],[grp-2] [usr-name]  // set many second-group to an user (replace).
groups [usr-name]  // list all group of an user.
id [usr-name]  // same.
userdel [usr-name]  // delete an user.
userdel -r [usr-name]  // same, delete home folder with it.
groupdel [grp-name]  // delete a group.

df  // print the disc space momory available.
df -h  // same, more readable.
df -t [ext4]  // print the space available for a type system.
df --total  // add a total line.
df -h | sort -k5  // order by purcent used.

du -h  // print the usage disc of system ("h" for mor readable).

cat [file-source] | tee [file-dest]  // "tee" write on a file AND print on console (use for save/historic).
tee -a [file]  // "a" for add to file (keep the previous data of file, write next to it).

clear  // erase the terminal printed.

help [cmd]  // print doc of a command (builtin, only for command interne).
[cmd] -h  // same.
[cmd] --help  // same.
man [cmd]  // same ("man" for manuel, documentation classic, use pagination).
info [cmd]  // same (more detailled than "man")

apropos [word-search]  // research a command who has a key-word on his manuel (for searching command).

history  // print the historic of command used.

whatis [cmd]  // print a description of a command.
whereis [cmd]  // print the emplacement of a command.

shutdown  // shut down the computer.
shutdown now  // same, (without cooldown verification, by default it has 1 minute to allow user to cancel).
shutdown 20:40  // same, but programme to shutdown at 20h40.
shutdown -c  // cancel a shutdown (has to use during the cooldown of 1 minute, or other).

uname -s  // print the name of kernel (noyau linux).
uname -n  // print the name of the PC (hostname).
uname -r  // print the vertion of kernel.
uname -v  // same, (more detailled).
uname -m  // print the name of material of PC (hardware ?).

hostname  // print the hote name.
hostname -i  // "i" for print the IP server.
hostname -a  // "a" for print the alias of hote.
hostname -A  // "A" for print full domaine name.

wget [url]  // download a file from web (like "apt install" but with a web adresse).
wget [url] -O [file-dest]  // "O" for precise in what file place it (can contain a path of file).
wget -r [url]  // "r" for recurcive (download all from the url).
wget -c [url]  // "c" for continue a previous wget download who was cut in midle.
wget -i [file]  // "i" for execute urls from a file.

uptime  // print details about the last start of PC (system and average charge).

file [file]  // return the type of a file.
file -k [file]  // same, more details.

sudo apt update  // maj of packages disponible (install nothing).
sudo apt upgrade  // installe maj of package already installed.
sudo apt install [package-name]  // install a new package.
dnf  // "dnf" replace "apt" for distro Fedora / Red Hat.

zip [file-zip-dest] [file(s)-to-zip]  // zip a file(s) into a zip file.
zip -r [fzd] [folder]  // same, "r" for zip a full folder (recursively).
unzip [file-zip]  // unzip a file.
unzip [file-zip] -d [path]  // same, "d" for send a destination.

tar -cvf [file-dest.tar] [file-A] [file-B]  // create an archive tar (c create, v verbose, f file).
tar -cvf [file-dest.tar] [folder]  // same with a full folder.
tar -czvf [file-dest.tar.gz] [folder]  // same but compressed.
tar -xvf [file-source.tar]  // "x" for extract (extract from a tar archive).
tar -xzvf [file-source.tar.gz]  // same for a compressed archive (tar.gz).
tar -xvf [file-source.tar] -C [path]  // extract on a path destination.
tar -tvf [file-source.tar]  // list contend of an archive (without extracting it).

exit  // quite from a session open.

sudo systemctl list-unit-files --type service --all  // list all task (like start the PC).
sudo systemctl [sub-cmd] --[option(s)]  // syntax.

watch ["cmd"]  // execute a cmd every 2 secondes.
watch -n 10 "ls /home"  // execute cmd ls every 10 secondes.
watch -d 10 "ls /home"  // "d" for hight-light the diference between two result.
