git commit   // send an edition of current branch.
git commit -a -m [comment]  // select all modif (-a) and add a comment (-m).

     v main*.
A -> B       // make B.


git checkout [nameBranchDestination]  // move to another branch.
git checkout -b [nameBranchDestination]  // -b for create a new one.

     v main local.
A -> B
     ^ second branch local*.


git merge [brancheWithEdition]  // make a new node with the current checkout (and merge the one send).

          v main*.      
A -> B -> C -> E  // merge from main using D as parameter. 
      \       /
       -> D ->
          ^ second branch local.

git rebase [brancheDest]  // add modification of current checkout into destination.

               v main (dest rebase).
A -> B -> C -> E
      \       
       -> D
          ^ branch edited*.

git log  // print ID of hashNode.

git checkout HEAD^  // comme back one node upper HEAD (in hashNode).
git checkout HEAD~2  // comme back 2 nodes upper HEAD (in hashNode).

git branch -f main HEAD~3  // force to replace current checkout has 3 node upper HEAD.

          v (move branch [firstParam] into [secondParam].
A -> B -> C
      \       
       -> D

git reset HEAD~1  // cancel the commit, and go back to node parent.

     v go back.
A -> B -> C

git revert HEAD  // make a commit with the modification of cancel last commit.

              v go next (roll back).
A -> B -> C > C'

git cherry-pick [commit1] [commit2] [...]  // move all commit send under the current checkout.

          v main*.   v branchs moved.
A -> B -> C -> D' -> E'
      \       
       -> D -> E

git rebase -i [branch]  // rebase interactif (open a vim menu to edit the order of rebase, and other option).

A -> B -> C
      \       
       -> C'
          ^ branch param re-create (by passing param B).

git commit --amend  // edit the last message commit.

          v checkout*.
A -> B -> C
      \       
       -> C'
          ^ new commit recommented.

git tag [tagName] [branch]  // mark a branch with a name tag (tag name can be use into other commend parameter).

     v (myTagName)
A -> B

git describe [branch]  // print a string as the path of the node (nameTag, node name and hash code).
